Eagle : Simple Testing Framework written in C++ for both C and C++

// How does it work?

* Have a testing class that you add tests too, and it runs them printing the result


// TIMELINE

* Get something working
* Refactor and redesign (use oop concepts)
* Macros


Example API:

====================================

// --- Unit test Example ---
int simple_test()
{
    int x = 5;
    x++;

    return (x == 6)
}

// --- Test Driver --- 
int main()
{

    Eagle E = Eagle();

    // Show detailed output
    E.verbose(true);

    E.addTest("simple_collection", "simple_test", &simple_test);
    E.addTest("simple_collection", "another_test", &another_test);

    // Run an individual Test
    E.runTest("simple_collection", "simple_test");

    // Run all tests in a given collection
    E.runCollection("simple_collection);

    // Run all tests in all collections
    E.runAll();

}
